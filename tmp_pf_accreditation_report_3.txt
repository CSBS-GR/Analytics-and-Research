SELECT * FROM
(SELECT A.YEAR, A.ACCOUNT, A.QUESTION_LABEL, A.RESPONSE
FROM TMP_PF_ALLYR_SECTIONI A
WHERE A.TOPIC_TITLE LIKE '1.%')
PIVOT
(
   MAX(RESPONSE)
   FOR QUESTION_LABEL IN ('Is your department independent, or part of a larger agency?' Q1_LARGE_AGENCY,
   'If part of a larger agency provide its name:' Q2_AGENCY_NAME,
   'To the best of your knowledge, in the next year is your department going to be affected by a state mandated reorganization?' Q3_AFFECTED_REORGANIZATION,
   'If yes, please explain.' Q4_EXPLAIN)
)
ORDER BY ACCOUNT
